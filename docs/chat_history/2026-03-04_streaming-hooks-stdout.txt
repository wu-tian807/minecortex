### 各 Hook 的触发点和 Payload：

| Hook | 粒度 | Payload | 触发位置 |
|------|------|---------|----------|
| `TurnStart` | 每轮开始 | `{ turn, eventCount }` | `brain.ts:327` - 主循环每轮处理事件前 |
| `TurnEnd` | 每轮结束 | `{ turn, aborted }` | `brain.ts:359` - 主循环每轮完成后 |
| `AssistantMessage` | 每条 LLM 响应 | `{ msg: LLMMessage, turn }` | `brain.ts:166` - 子 agent 循环中 |
| `ToolCall` | 每次工具调用前 | `{ name, args, toolCall }` | `brain.ts:184`, `brain.ts:389` |
| `ToolResult` | 每次工具执行完 | `{ name, result, durationMs }` | `brain.ts:187`, `brain.ts:409` |

### 粒度评估：

**覆盖的部分：**
- ✅ Turn 级别的开始/结束
- ✅ 工具调用/结果级别
- ✅ Assistant 消息级别

**缺少的粒度（可能需要的）：**
- ❌ `LLMRequest` / `LLMResponse` — LLM 调用前后
- ❌ `EventReceived` — 事件到达时
- ❌ `BrainStart` / `BrainStop` — brain 整体生命周期
- ❌ `PromptBuilt` — prompt 构建完成时
- ❌ `SessionSaved` / `SessionLoaded` — session 持久化相关
- ❌ `Error` — 错误处理

当前的 hooks 是一个 **中等粒度** 的设计，主要聚焦于 **turn 循环** 和 **工具执行** 两个维度。如果需要更细粒度的监控（如 LLM streaming、token 计数、retry 等），目前的 hook 系统还不支持。

import * as readline from "node:readline";
import { join } from "node:path";
import type { Event, EventSource, SourceContext } from "../src/core/types.js";
import { parseCommand } from "../src/core/command-parser.js";
import { QARecorder } from "../src/session/qa-recorder.js";
import { HookEvent } from "../src/hooks/types.js";

export default function create(ctx: SourceContext): EventSource {
  let rl: readline.Interface | null = null;
  const qa = new QARecorder(join(ctx.brainDir, "logs"));
  const disposers: Array<() => void> = [];

  return {
    name: "stdin",

    start(emit: (event: Event) => void) {
      rl = readline.createInterface({ input: process.stdin });

      // ─── Streaming output state ───
      let streaming = false;
      let streamedText = false;

      // TurnStart: prepare for streaming
      disposers.push(ctx.hooks.on(HookEvent.TurnStart, () => {
        streaming = true;
        streamedText = false;
      }));

      // StreamChunk: real-time output
      disposers.push(ctx.hooks.on(HookEvent.StreamChunk, ({ chunk }) => {
        if (!streaming) return;

        if (chunk.type === "text") {
          process.stdout.write(chunk.text);
          streamedText = true;
        }
      }));

      // ToolCall: pause streaming, show tool indicator
      disposers.push(ctx.hooks.on(HookEvent.ToolCall, ({ name }) => {
        if (streamedText) {
          process.stdout.write("\n");
          streamedText = false;
        }
        process.stdout.write(`\x1b[90m▶ ${name}\x1b[0m `);
      }));

      // ToolResult: show completion indicator
      disposers.push(ctx.hooks.on(HookEvent.ToolResult, ({ durationMs }) => {
        process.stdout.write(`\x1b[90m(${durationMs}ms)\x1b[0m\n`);
      }));

      // TurnEnd: finalize output
      disposers.push(ctx.hooks.on(HookEvent.TurnEnd, () => {
        if (streamedText) {
          process.stdout.write("\n");
        }
        streaming = false;
        streamedText = false;
      }));

      // AssistantMessage: record to QA log (after full message assembled)
      disposers.push(ctx.hooks.on(HookEvent.AssistantMessage, ({ msg }) => {
        const text = typeof msg.content === "string" ? msg.content : "";
        if (text) {
          qa.recordAssistant(text).catch(() => {});
        }
      }));

      rl.on("line", (line) => {
        const trimmed = line.trim();
        if (trimmed.length === 0) return;

        if (trimmed.startsWith("/")) {
          const cmd = parseCommand(trimmed);
          if (cmd && ctx.onCommand) {
            ctx.onCommand(cmd.toolName, cmd.args, cmd.target);
            return;
          }
        }

        qa.recordUser(trimmed).catch(() => {});

        emit({
          source: "stdin",
          type: "message",
          payload: { text: trimmed },
          ts: Date.now(),
        });
      });

      rl.on("close", () => {
        console.log("[stdin] 输入流已关闭");
        rl = null;
      });

      console.log("[stdin] 订阅已启动，等待输入...");
    },

    stop() {
      for (const dispose of disposers) {
        dispose();
      }
      disposers.length = 0;
      rl?.close();
      rl = null;
      qa.close();
    },
  };
}

export default function create(ctx: SourceContext): EventSource {
  const unsubs: (() => void)[] = [];
  const qa = new QARecorder(join(ctx.brainDir, "logs"));

  return {
    name: "stdout",

    start(_emit: (event: Event) => void) {
      if (!brainHasStdin(ctx)) return;

      // ─── Streaming state ───
      let streaming = false;
      let streamedText = "";
      let pendingToolCalls = false;

      // TurnStart: prepare for streaming
      unsubs.push(ctx.hooks.on(HookEvent.TurnStart, () => {
        streaming = true;
        streamedText = "";
        pendingToolCalls = false;
      }));

      // StreamChunk: real-time text output
      unsubs.push(ctx.hooks.on(HookEvent.StreamChunk, ({ chunk }) => {
        if (!streaming) return;

        if (chunk.type === "text") {
          process.stdout.write(chunk.text);
          streamedText += chunk.text;
        }
      }));

      // ToolCall: show tool indicator, pause text streaming
      unsubs.push(ctx.hooks.on(HookEvent.ToolCall, ({ name, args }) => {
        if (streamedText && !streamedText.endsWith("\n")) {
          process.stdout.write("\n");
        }
        const argsPreview = JSON.stringify(args).slice(0, 80);
        process.stdout.write(`\x1b[90m  → ${name}(${argsPreview})\x1b[0m\n`);
        pendingToolCalls = true;
      }));

      // ToolResult: show result preview
      unsubs.push(ctx.hooks.on(HookEvent.ToolResult, ({ name, result, durationMs }) => {
        const resultStr = typeof result === "string" ? result : JSON.stringify(result);
        const preview = resultStr.slice(0, 200);
        const suffix = resultStr.length > 200 ? "..." : "";
        process.stdout.write(`\x1b[90m  ← ${name} (${durationMs}ms): ${preview}${suffix}\x1b[0m\n`);
        qa.recordToolResult(name, resultStr, durationMs).catch(() => {});
      }));

      // TurnEnd: finalize and record
      unsubs.push(ctx.hooks.on(HookEvent.TurnEnd, () => {
        if (streaming && streamedText) {
          const clean = streamedText.replace(/<thinking>[\s\S]*?<\/thinking>\n?/g, "").trim();
          if (clean) {
            qa.recordAssistant(clean).catch(() => {});
          }
          if (!streamedText.endsWith("\n") && !pendingToolCalls) {
            process.stdout.write("\n");
          }
        }
        streaming = false;
        streamedText = "";
        pendingToolCalls = false;
